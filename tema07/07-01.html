Módulos y Paquetes

Teoría y Ejemplos

1. ¿Qué son los Módulos?

Un módulo es un archivo Python (.py) que contiene definiciones y declaraciones.

¿Por qué usar módulos?

  • Organización: Divide el código en partes lógicas
  • Reutilización: Usa el mismo código en múltiples proyectos
  • Namespace: Evita conflictos de nombres
  • Mantenibilidad: Código más fácil de mantener
  • Colaboración: Diferentes personas trabajan en diferentes módulos

Tipos de módulos:

  1. Módulos incorporados (built-in): Ya vienen con Python
  2. Módulos de la biblioteca estándar: Incluidos con Python
  3. Módulos de terceros: Instalados con pip
  4. Módulos propios: Los que tú creas

2. Importar Módulos

Hay varias formas de importar módulos en Python.

2.1. Importar un Módulo Completo

# Importar el módulo math completo
import math

# Usar funciones del módulo con la sintaxis: modulo.funcion()
print(f"Pi: {math.pi}")
print(f"Raíz cuadrada de 16: {math.sqrt(16)}")
print(f"Seno de 90 grados: {math.sin(math.radians(90))}")
print(f"Factorial de 5: {math.factorial(5)}")
# Importar múltiples módulos
import random
import datetime

print(f"Número aleatorio: {random.randint(1, 100)}")
print(f"Fecha actual: {datetime.datetime.now()}")

3. Módulos de la Biblioteca Estándar

Python viene con una "batería incluida" - muchos módulos útiles ya instalados.

3.1. math - Operaciones Matemáticas

import math

# Constantes
print(f"π = {math.pi}")
print(f"e = {math.e}")
print(f"τ = {math.tau}")  # tau = 2π
print(f"∞ = {math.inf}")

print("\n--- Funciones básicas ---")
# Redondeo y valor absoluto
print(f"ceil(4.3) = {math.ceil(4.3)}")    # Redondeo hacia arriba
print(f"floor(4.7) = {math.floor(4.7)}")  # Redondeo hacia abajo
print(f"fabs(-5) = {math.fabs(-5)}")      # Valor absoluto

print("\n--- Potencias y raíces ---")
print(f"sqrt(16) = {math.sqrt(16)}")
print(f"pow(2, 3) = {math.pow(2, 3)}")
print(f"exp(2) = {math.exp(2)}")  # e^2
print(f"log(100, 10) = {math.log(100, 10)}")  # Logaritmo base 10

print("\n--- Trigonometría ---")
print(f"sin(π/2) = {math.sin(math.pi/2)}")
print(f"cos(π) = {math.cos(math.pi)}")
print(f"tan(π/4) = {math.tan(math.pi/4)}")

# Convertir grados a radianes
print(f"sin(90°) = {math.sin(math.radians(90))}")

3.2. random - Números Aleatorios

import random

# Número aleatorio flotante entre 0 y 1
print(f"random(): {random.random()}")
# Número entero aleatorio en un rango
print(f"\nrandint(1, 10): {random.randint(1, 10)}")
print(f"randrange(0, 100, 5): {random.randrange(0, 100, 5)}")  # De 0 a 100, de 5 en 5
# Número flotante en un rango
print(f"\nuniform(1.0, 10.0): {random.uniform(1.0, 10.0)}")
# Elegir elemento aleatorio
colores = ["rojo", "verde", "azul", "amarillo"]
print(f"\nchoice(colores): {random.choice(colores)}")
# Elegir múltiples elementos (con repetición)
print(f"choices(colores, k=3): {random.choices(colores, k=3)}")
# Elegir múltiples elementos (sin repetición)
print(f"sample(colores, k=2): {random.sample(colores, k=2)}")
# Mezclar una lista
numeros = [1, 2, 3, 4, 5]
random.shuffle(numeros)
print(f"\nLista mezclada: {numeros}")
# Ejemplo práctico: Lanzar dados
def lanzar_dado():
    return random.randint(1, 6)

def lanzar_dos_dados():
    return lanzar_dado(), lanzar_dado()

print("Lanzar un dado:")
for _ in range(5):
    print(f"  🎲 {lanzar_dado()}")

print("\nLanzar dos dados:")
for _ in range(5):
    d1, d2 = lanzar_dos_dados()
    print(f"  🎲🎲 {d1} + {d2} = {d1 + d2}")

3.7. collections - Estructuras de Datos Especializadas

from collections import Counter, defaultdict, namedtuple, deque

# Counter - Contar elementos
palabras = ["python", "java", "python", "c++", "python", "java"]
contador = Counter(palabras)
print("Counter:")
print(f"  Conteo: {contador}")
print(f"  Más común: {contador.most_common(2)}")
# defaultdict - Diccionario con valor por defecto
grupos = defaultdict(list)
grupos['frutas'].append('manzana')
grupos['frutas'].append('pera')
grupos['verduras'].append('lechuga')
print("\ndefaultdict:")
print(f"  {dict(grupos)}")
# namedtuple - Tupla con nombres
Punto = namedtuple('Punto', ['x', 'y'])
p1 = Punto(10, 20)
print("\nnamedtuple:")
print(f"  Punto: {p1}")
print(f"  x: {p1.x}, y: {p1.y}")
# deque - Cola de doble extremo
cola = deque([1, 2, 3])
cola.append(4)      # Agregar al final
cola.appendleft(0)  # Agregar al inicio
print("\ndeque:")
print(f"  Cola: {cola}")
print(f"  Pop derecha: {cola.pop()}")
print(f"  Pop izquierda: {cola.popleft()}")
print(f"  Cola final: {cola}")

3.8. itertools - Herramientas de Iteración

import itertools

# combinations - Combinaciones
letras = ['A', 'B', 'C']
combinaciones = list(itertools.combinations(letras, 2))
print(f"Combinaciones de 2: {combinaciones}")
# permutations - Permutaciones
permutaciones = list(itertools.permutations(letras, 2))
print(f"Permutaciones de 2: {permutaciones}")
# product - Producto cartesiano
colores = ['rojo', 'azul']
tallas = ['S', 'M', 'L']
producto = list(itertools.product(colores, tallas))
print(f"\nProducto cartesiano: {producto}")
# chain - Concatenar iterables
lista1 = [1, 2, 3]
lista2 = [4, 5, 6]
concatenado = list(itertools.chain(lista1, lista2))
print(f"\nConcatenado: {concatenado}")
# count - Contador infinito
contador = itertools.count(start=10, step=2)
print("\nPrimeros 5 números del contador:")
for _ in range(5):
    print(f"  {next(contador)}")
# cycle - Ciclar sobre una secuencia
colores_ciclo = itertools.cycle(['rojo', 'verde', 'azul'])
print("\nPrimeros 7 del ciclo:")
for _ in range(7):
    print(f"  {next(colores_ciclo)}")

4. Crear Módulos Propios

Puedes crear tus propios módulos para organizar tu código.

4.1. Estructura Básica de un Módulo

Archivo: calculadora.py

"""
Módulo de operaciones matemáticas básicas.
"""

# Variables del módulo
VERSION = "1.0"
AUTOR = "Tu Nombre"

# Funciones del módulo
def sumar(a, b):
    """Suma dos números."""
    return a + b

def restar(a, b):
    """Resta dos números."""
    return a - b

def multiplicar(a, b):
    """Multiplica dos números."""
    return a * b

def dividir(a, b):
    """Divide dos números."""
    if b == 0:
        raise ValueError("No se puede dividir por cero")
    return a / b

# Código que se ejecuta solo si el módulo se ejecuta directamente
if __name__ == "__main__":
    # Pruebas del módulo
    print("Pruebas del módulo calculadora")
    print(f"5 + 3 = {sumar(5, 3)}")
    print(f"10 - 4 = {restar(10, 4)}")
    print(f"6 * 7 = {multiplicar(6, 7)}")
    print(f"15 / 3 = {dividir(15, 3)}")

Usar el módulo:

4.3. Ejemplo de Módulo Completo

Archivo: utilidades_texto.py

"""
Utilidades para procesamiento de texto.
"""

def contar_palabras(texto):
    """Cuenta las palabras en un texto."""
    return len(texto.split())

def contar_vocales(texto):
    """Cuenta las vocales en un texto."""
    vocales = 'aeiouAEIOUáéíóúÁÉÍÓÚ'
    return sum(1 for char in texto if char in vocales)

def invertir_texto(texto):
    """Invierte un texto."""
    return texto[::-1]

def es_palindromo(texto):
    """Verifica si un texto es palíndromo."""
    texto_limpio = ''.join(texto.lower().split())
    return texto_limpio == texto_limpio[::-1]

def capitalizar_palabras(texto):
    """Capitaliza la primera letra de cada palabra."""
    return texto.title()

class AnalizadorTexto:
    """Clase para analizar texto."""
    
    def __init__(self, texto):
        self.texto = texto
    
    def analizar(self):
        """Retorna un análisis completo del texto."""
        return {
            'caracteres': len(self.texto),
            'palabras': contar_palabras(self.texto),
            'vocales': contar_vocales(self.texto),
            'es_palindromo': es_palindromo(self.texto)
        }

if __name__ == "__main__":
    # Pruebas
    texto = "Python es genial"
    print(f"Palabras: {contar_palabras(texto)}")
    print(f"Vocales: {contar_vocales(texto)}")
    print(f"Invertido: {invertir_texto(texto)}")

5. Paquetes (Packages)

Un paquete es una colección de módulos organizados en directorios.

5.1. Estructura de un Paquete

mi_paquete/
│
├── __init__.py          # Hace que el directorio sea un paquete
├── modulo1.py
├── modulo2.py
│
└── subpaquete/
    ├── __init__.py
    ├── modulo3.py
    └── modulo4.py

Archivo: mi_paquete/__init__.py

"""
Mi Paquete - Colección de utilidades.
"""

# Versión del paquete
__version__ = "1.0.0"
__author__ = "Tu Nombre"

# Importar funciones principales para facilitar el acceso
from .modulo1 import funcion_importante
from .modulo2 import otra_funcion

# Definir qué se exporta con "from mi_paquete import *"
__all__ = ['funcion_importante', 'otra_funcion']

"""Utilidades matemáticas."""

def factorial(n):
    if n <= 1:
        return 1
    return n * factorial(n - 1)

def es_primo(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

"""Validación de emails."""

def validar_email(email):

Sistema de Gestión de Biblioteca

# Simulación de un sistema de biblioteca usando módulos

import json
from datetime import datetime, timedelta
from collections import defaultdict

class Libro:
    """Representa un libro en la biblioteca."""
    
    def __init__(self, isbn, titulo, autor, año):
        self.isbn = isbn
        self.titulo = titulo
        self.autor = autor
        self.año = año
        self.prestado = False
        self.fecha_prestamo = None
    
    def to_dict(self):
        """Convierte el libro a diccionario."""
        return {
            'isbn': self.isbn,
            'titulo': self.titulo,
            'autor': self.autor,
            'año': self.año,
            'prestado': self.prestado
        }
    
    def __str__(self):
        estado = "Prestado" if self.prestado else "Disponible"
        return f"{self.titulo} - {self.autor} ({self.año}) [{estado}]"

class Biblioteca:
    """Gestiona una colección de libros."""
    
    def __init__(self):
        self.libros = {}
        self.prestamos = defaultdict(list)
    
    def agregar_libro(self, libro):
        """Agrega un libro a la biblioteca."""
        self.libros[libro.isbn] = libro
        print(f"Libro agregado: {libro.titulo}")
    
    def buscar_por_titulo(self, titulo):
        """Busca libros por título."""
        resultados = []
        for libro in self.libros.values():
            if titulo.lower() in libro.titulo.lower():
                resultados.append(libro)
        return resultados
    
    def buscar_por_autor(self, autor):
        """Busca libros por autor."""
        resultados = []
        for libro in self.libros.values():
            if autor.lower() in libro.autor.lower():
                resultados.append(libro)
        return resultados
    
    def prestar_libro(self, isbn, usuario):
        """Presta un libro a un usuario."""
        if isbn not in self.libros:
            return False, "Libro no encontrado"
        
        libro = self.libros[isbn]
        if libro.prestado:
            return False, "Libro ya prestado"
        
        libro.prestado = True
        libro.fecha_prestamo = datetime.now()
        self.prestamos[usuario].append({
            'isbn': isbn,
            'fecha': libro.fecha_prestamo
        })
        
        return True, f"Libro prestado a {usuario}"
    
    def devolver_libro(self, isbn):
        """Devuelve un libro prestado."""
        if isbn not in self.libros:
            return False, "Libro no encontrado"
        
        libro = self.libros[isbn]
        if not libro.prestado:
            return False, "Libro no estaba prestado"
        
        libro.prestado = False
        libro.fecha_prestamo = None
        
        return True, "Libro devuelto exitosamente"
    
    def listar_disponibles(self):
        """Lista todos los libros disponibles."""
        return [libro for libro in self.libros.values() if not libro.prestado]
    
    def estadisticas(self):
        """Muestra estadísticas de la biblioteca."""
        total = len(self.libros)
        prestados = sum(1 for libro in self.libros.values() if libro.prestado)
        disponibles = total - prestados
        
        return {
            'total': total,
            'prestados': prestados,
            'disponibles': disponibles
        }

# Usar el sistema
biblioteca = Biblioteca()

# Agregar libros
libro1 = Libro("978-1", "Python para Todos", "Juan Pérez", 2020)
libro2 = Libro("978-2", "Aprende Python", "Ana García", 2021)
libro3 = Libro("978-3", "Python Avanzado", "Juan Pérez", 2022)

biblioteca.agregar_libro(libro1)
biblioteca.agregar_libro(libro2)
biblioteca.agregar_libro(libro3)

# Buscar libros
print("\nBuscar 'Python':")
resultados = biblioteca.buscar_por_titulo("Python")
for libro in resultados:
    print(f"  {libro}")

# Prestar libro
print("\nPrestar libro:")
exito, mensaje = biblioteca.prestar_libro("978-1", "Carlos")
print(f"  {mensaje}")

# Estadísticas
print("\nEstadísticas:")
stats = biblioteca.estadisticas()
print(f"  Total: {stats['total']}")
print(f"  Prestados: {stats['prestados']}")
print(f"  Disponibles: {stats['disponibles']}")

# Listar disponibles
print("\nLibros disponibles:")
for libro in biblioteca.listar_disponibles():
    print(f"  {libro}")
